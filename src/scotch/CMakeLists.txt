macro(ADD_EXEC NAME)
    add_executable(${NAME} ${NAME}.c)
    target_link_libraries(${NAME} PRIVATE scotch)
    target_include_directories(${NAME} PRIVATE ../libscotch ${SCOTCH_INC_DIR} .)
endmacro()
